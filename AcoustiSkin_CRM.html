    function attachOrderListeners() {

      document.getElementById('addOrderItem')?.addEventListener('click', () => {

        const product = document.getElementById('orderProduct').value;

        const size = document.getElementById('orderSize').value;

        const color = document.getElementById('orderColor').value;

        const custom = document.getElementById('orderCustom').value;

        const qty = parseInt(document.getElementById('orderQty').value) || 1;

        const price = parseFloat(document.getElementById('orderPrice').value) || 0;

        if (product && price > 0) {

          orderItems.push({ product, size, color, custom, qty, price: price * qty });

          document.getElementById('orderProduct').value = '';

          document.getElementById('orderCustom').value = '';

          document.getElementById('orderPrice').value = '';

          updateOrderDisplay();

        }

      });

      document.getElementById('processOrderBtn')?.addEventListener('click', () => {

        if (orderItems.length === 0) {

          showNotification('Add items first!', 'bg-red-100 text-red-700');

          return;

        }

        showNotification(`Order processed! Total: $${orderItems.reduce((s,i)=>s+i.price,0).toFixed(2)}`, 'bg-green-100 text-green-700');

        orderItems = [];

        renderOrderForm();

      });

    }

            <p class="font-medium">${item.qty}x ${item.product} (${item.size}, ${item.color})</p>

          <p class="font-medium">$${item.price.toFixed(2)}</p>

    function renderCalendar() {

      const view = document.getElementById('calendar-view');

      const today = new Date();

      const year = today.getFullYear();

      const month = today.getMonth();

      const firstDay = new Date(year, month, 1).getDay();

      const daysInMonth = new Date(year, month + 1, 0).getDate();

      let html = `

        <div class="max-w-7xl mx-auto">

          <h2 class="text-2xl font-extrabold text-gray-900 mb-6">Calendar - ${today.toLocaleString('default', { month: 'long' })} ${year}</h2>

          <div class="grid grid-cols-7 gap-2 text-center font-semibold text-sm">

            <div>Sun</div><div>Mon</div><div>Tue</div><div>Wed</div><div>Thu</div><div>Fri</div><div>Sat</div>

          </div>

          <div class="grid grid-cols-7 gap-2 mt-2">`;

      for (let i = 0; i < firstDay; i++) html += `<div></div>`;

      for (let day = 1; day <= daysInMonth; day++) {

        const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`;

        const dayAppts = appointments.filter(a => a.date === dateStr);

        const dayTasks = tasks.filter(t => t.due === dateStr);

        html += `

          <div class="border rounded-lg p-2 min-h-24 bg-white ${day === today.getDate() ? 'ring-2 ring-indigo-600' : ''}">

            <div class="font-semibold text-sm">${day}</div>

            ${dayAppts.map(a => `<div class="text-xs bg-blue-100 p-1 rounded mt-1">${a.title}</div>`).join('')}

            ${dayTasks.map(t => `<div class="text-xs bg-yellow-100 p-1 rounded mt-1">${t.title}</div>`).join('')}

          </div>`;

      }

      html += `</div></div>`;

      view.innerHTML = html;

    }
